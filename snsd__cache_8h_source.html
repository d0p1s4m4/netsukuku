66/*
67 * snsd_node, snsd_service, snsd_prio
68 *
69 * They are three linked list. They are all orthogonal to each other.
70 * The snsd_node llist is inside each snsd_prio struct which is inside each
71 * snsd_service struct:
72 * || service X <-> service Y <-> service Z <-> ... ||
73 * | | |
74 * V V V
75 * snsd_prio_1-->node snsd_prio_1-->node ...-->...
76 * | |
77 * V V
78 * snsd_prio_2-->node ...-->node
79 * |
80 * V
81 * ...-->node
82 *
83 * Using this schema, we don't have to sort them, ever. The nodes are already
84 * grouped by service and in each service by priority.
85 *
86 * These llist are directly embedded in the andna_cache, lcl_cache and
87 * rh_cache.
88 *
89 * The andna_cache keeps all the SNSD nodes associated to the registered
90 * hostname. The andna_cache doesn't need `snsd_node->pubkey'.
91 *
92 * The rh_cache stores only records which are of the SNSD_NODE_IP type.
93 *
94 * When the lcl_cache is saved, its snsd llist is discarded because it is
95 * loaded each time from the /etc/netsukuku/snsd_nodes file.
96 */
97struct snsd_node {
98 LLIST_HDR(struct snsd_node);
99
100 u_int record[MAX_IP_INT]; /* It can be the IP or the md5
101 hash of the hname of the
102 SNSD node */
103 RSA *pubkey; /* pubkey of the snsd_node */
104 char flags; /* This will tell us what
105 `record' is */
106
107 u_char weight;
108};

142struct snsd_node_llist_hdr {
143 u_short count; /* # of snsd_node structs packed
144 in the body */
145} _PACKED_;
146INT_INFO snsd_node_llist_hdr_iinfo = { 1, {INT_TYPE_16BIT}, {0}, {1} };
147
148/*
149 * the body of the pkt is:
150 *
151 * struct snsd_node_pack {
152 * u_int record[MAX_IP_INT];
153 * char flags;
154 * u_char weight;
155 * } pack[hdr.nodes];
156 */
157#define SNSD_NODE_LLIST_PACK_SZ(head) (list_count((head))*SNSD_NODE_PACK_SZ \
158 + sizeof(struct snsd_node_llist_hdr))
159
160struct snsd_prio_llist_hdr {
161 u_short count; /* number of structs packed in
162 the body */
163} _PACKED_;
164INT_INFO snsd_prio_llist_hdr_iinfo = { 1, {INT_TYPE_16BIT}, {0}, {1} };
165
166/*
167 * the body is:
168 *
169 * snsd_prio_pack {
170 * u_char prio;
171 * char snsd_node_llist_pack[SNSD_NODE_LLIST_PACK_SZ];
172 * } pack[hdr.count];
173 */
174#define SNSD_PRIO_LLIST_PACK_SZ(head) \
175({ \
176 snsd_prio *_p=(head); \
177 int _priosz=0; \
178 \
179 list_for(_p) { \
180 _priosz+=SNSD_NODE_LLIST_PACK_SZ(_p->node); \
181 _priosz+=SNSD_PRIO_PACK_SZ; \
182 } \
183 _priosz+=sizeof(struct snsd_prio_llist_hdr); \
184 _priosz; \
185})
186
187
188struct snsd_service_llist_hdr {
189 u_short count;
190} _PACKED_;
191INT_INFO snsd_service_llist_hdr_iinfo = { 1, {INT_TYPE_16BIT}, {0}, {1} };
192
193/*
194 * the body is:
195 * u_short service;
196 * u_char proto;
197 * char snsd_prio_llist_pack[SNSD_PRIO_LLIST_PACK_SZ];
198 */
199#define SNSD_SERVICE_LLIST_PACK_SZ(head) \
200({ \
201 snsd_service *_s=(head); \
202 int _srvsz=0; \
203 if(_s) { \
204 list_for(_s) { \
205 _srvsz+=SNSD_PRIO_LLIST_PACK_SZ(_s->prio); \
206 _srvsz+=SNSD_SERVICE_PACK_SZ; \
207 } \
208 _srvsz+=sizeof(struct snsd_service_llist_hdr); \
209 } \
210 _srvsz; \
211})
212
213#define SNSD_SERVICE_SINGLE_PACK_SZ(head) \
214({ SNSD_SERVICE_PACK_SZ + \
215 SNSD_PRIO_LLIST_PACK_SZ((head)->prio); \
216})
217
218#define SNSD_SERVICE_MAX_PACK_SZ \
219( ( (SNSD_NODE_PACK_SZ + SNSD_PRIO_PACK_SZ) * \
220 (SNSD_MAX_REC_SERV) ) + \
221 SNSD_SERVICE_PACK_SZ + \
222 sizeof(struct snsd_prio_llist_hdr) + \
223 sizeof(struct snsd_service_llist_hdr) \
224)
225
226#define SNSD_SERVICE_MAX_LLIST_PACK_SZ \
227(( SNSD_NODE_PACK_SZ + SNSD_PRIO_PACK_SZ + SNSD_SERVICE_PACK_SZ + \
228 sizeof(struct snsd_prio_llist_hdr))*SNSD_MAX_RECORDS + \
229 sizeof(struct snsd_service_llist_hdr) \
230)
