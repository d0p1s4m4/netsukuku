60/*\
61 * *** Map notes ***
62 *
63 * The map is an array of MAXGROUPNODE map_node structs. It is a generic map
64 * and it is used to keep the qspn_map, the internal map and the external map.
65 * The position in the map of each struct corresponds to its relative ip.
66 * For example, if the map goes from 192.128.1.0 to 192.128.3.0, the map will
67 * have 512 structs, the first one will correspond to 192.168.1.0, the 50th to
68 * 192.168.1.50 and so on.
69 * Note: because MAXGROUPNODE is 256, we can use an u_char for the index of the
70 * array.
71 *
72\*/
73
74/* map_rnode is what map_node.r_node points to. (read struct map_node below) */
75typedef struct {
76 int *r_node; /*It's the pointer to the struct of the
77 r_node in the map */
78 u_int trtt;
79 /*
80 * node <-> root_node total rtt: The rtt to reach the root_node
81 * starting from the node which uses this rnode (in millisec).
82 * Cuz I've explained it in such a bad way I make an example:
83 * map_node node_A; From node_A "node_A.links"th routes to the root_node
84 * start. So I have "node_A.links"th node_A.r_node[s], each of them is a
85 * different route to reach the root_node.
86 * With the node_A.r_node[route_number_to_follow].trtt I can get the rtt
87 * needed to reach the root_node starting from the node_A using the
88 * route_number_to_follow. Gotcha? I hope so.
89 * Note: The trtt is mainly used to sort the routes
90 */
91} map_rnode;

105/*
106 * ****) The qspn int_map (****
107 *
108 * - map_node.r_node points to the r_node of the root_node to be used as
109 * gateway to reach map_node. So map_node.r_node stores only the gateway
110 * needed to reach map_node from the root_node.
111 * The only execption is the root_node itself. The root_node's
112 * map_node.r_node keeps all its rnodes as a normal (non qspn) map would.
113 *
114 * The only exception is the root_node. Its rnodes have a different meaning:
115 * they are its effective rnodes, so each map_node.r_node points to the node
116 * which is the real rnode of the root_node.
117 * The root_node at level 0 may have also rnode of a different gnode
118 * (it is a border node).
119 * To store these external rnodes in root_node.r_node[x], the
120 * root_node.r_node[x].r_node will point to the relative ext_rnode struct
121 * (see gmap.h) and the MAP_GNODE | MAP_ERNODE flags will be set in
122 * root_node.r_node[x].flags.
123 * The rnodes of the root_node of 0 level are updated by the radar(),
124 * instead the root_nodes of greater levels are updated by the qspn.
125 */
126typedef struct {
127 u_short flags;
128 u_int brdcast; /*Pkt_id of the last brdcast_pkt sent by this node */
129 u_short links; /*Number of r_nodes */
130 map_rnode *r_node; /*These structs will be kept in ascending
131 order considering their rnode_t.rtt */
132} map_node;
