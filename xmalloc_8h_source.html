42/*
43 * xfree
44 *
45 * It calls _xfree(__pptr) and then sets `__pptr' to 0.
46 * It is safe, you can use it also with expressions:
47 * xfree(a++);
48 */
49#define xfree(__pptr) \
50do{ \
51 char **_p=(char **)&(__pptr); \
52 _xfree(*_p); \
53 *_p=0; \
54}while(0)
