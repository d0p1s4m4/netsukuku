56/*
57 * The free_nodes pkt is used to send the list of all the free/not used
58 * nodes/gnodes present at level fn_hdr.level.
59 * If level is == 0 then the pkt contains the list of free nodes of the fn_hdr.gid
60 * gnode.
61 * If level is > 0 then it contains the list of free gnodes which are inside
62 * the gnode with `fn_hdr.gid' gid of the `fn_hdr.level'th level. So the free gnodes
63 * are part of the (fn_hdr.level - 1)th level.
64 */
65struct free_nodes_hdr {
66 u_char max_levels; /* How many levels we are managing */
67
68 uint32_t ipstart[MAX_IP_INT]; /* The ipstart of the gnode */
69 u_char level; /* The level where the gnode belongs */
70 u_char gid; /* The gnode id */
71 u_char nodes; /* The number of free (g)nodes - 1 */
72 uint32_t join_rate; /* Join_rate of `level' */
73} _PACKED_;
74INT_INFO free_nodes_hdr_iinfo = { 1, {INT_TYPE_32BIT},
75{sizeof(struct free_nodes_hdr) - sizeof(uint32_t)}
76,
77{1}
78};
79
80#define FREE_NODES_SZ(nodes) (sizeof(struct free_nodes_hdr) +\
81 (sizeof(u_char) * (nodes)))

83/*
84 * the free_nodes block is:
85 *
86 * u_char free_nodes[fn_hdr.nodes/8]; The bit x inside `free_nodes'
87 * tells if the node x, i.e.
88 * int_map[x], is up or not.
89 * If the bit is set to 1, it
90 * is.
91 *
92 * The free_nodes pkt is:
93 * fn_hdr;
94 * fn_block;
95 */
96
97/*
98 * the qspn_round pkt is:
99 * u_char max_levels;
100 * int qspn_id[max_levels]; the qspn_id of the last qspn_round for each
101 * fn_hdr.max_levels level
102 * struct timeval qtime[max_levels]; qspn round time: how many seconds passed away
103 * since the previous qspn round. There's a qtime
104 * for each fn_hdr.max_levels level
105 * u_int gcount[GCOUNT_LEVELS]; current qspn_gnode_count.
106 */
107/* Note: for this int_info we are considering the timeval array as one int
108 * with `max_levels'*2 members */
109INT_INFO qspn_round_pkt_iinfo = { 3,
110 {INT_TYPE_32BIT, INT_TYPE_32BIT, INT_TYPE_32BIT}
111 ,
112 {sizeof(char), IINFO_DYNAMIC_VALUE, IINFO_DYNAMIC_VALUE},
113 {IINFO_DYNAMIC_VALUE, IINFO_DYNAMIC_VALUE, GCOUNT_LEVELS}
114};
