65/*
66 * * * * Levels notes * * *
67 *
68 * These are the levels of the external_map. Note that the 0 level is never used
69 * for the ext_map because it corresponds to the internal map. Btw the 0 level is
70 * counted so the number of LEVELS includes it too.
71 * But we have to add another extra level: the last exiled level. It is also never
72 * used but it is vital, cause, its gnode 0 includes the entire Netsukuku, the other
73 * gnodes aren't used, it is a mere symbol. We call it the unity level.
74 *
75 * All the structs/arrays related to the external map, and the ext_map itself, don't
76 * use the EXTRA_LEVELS, thus, they lack of the zero level. To retrieve the position
77 * in the array from the level the _EL macro must be used. In other words:
78 * because the arrays goes from 0 to n-1 we refer to the levels as the arrays,
79 * so the level 1 is the level 0, the level 2 is the level 1, and so on.
80 * These arrays/structs are: quadg.gnode, rblock, ext_map, qspn_gnode_count.
81 */
82#define ZERO_LEVEL 1
83#define UNITY_LEVEL 1
84#define EXTRA_LEVELS (ZERO_LEVEL + UNITY_LEVEL)
85/* To use the right level. */
86#define _EL(level) ((level)-1)
87/* And to restore it. */
88#define _NL(level) ((level)+1)
89
90/*
91 * Using MAXGROUPNODE = 2^8; IPV4_LEVELS = 3; ips = 2^32;
92 * ips/(MAXGROUPNODE^IPV4_LEVELS) == 256;
93 * If we use IPV4_LEVELS = 3, we almost cover all the ips, but the division gives
94 * 256. So there are only 256 groups in the last level (3), in fact:
95 * ips/(256 * (MAXGROUPNODE^3)) == 1
96 * And to include them we use the unity level, thus IPV4_LEVELS is equal to 3+1.
97 * This means that the unity level is the one which has only one group node which includes
98 * the entire network.
99 * Sadly we cannot use all this ips, because there are the banned classes (MULTICAST,
100 * ZERONET), the kernel will sput on us.
101 *
102 * For the ipv6 we have IPV6_LEVELS = 16, ips = 2^128; so:
103 * ips/(MAXGROUPNODE^16) == 1
104 */
105#define IPV4_LEVELS (2+EXTRA_LEVELS)
106
107#define IPV6_LEVELS (14+EXTRA_LEVELS)
108
109#define MAX_LEVELS IPV6_LEVELS
110#ifdef DEBUG
111#define GET_LEVELS(family) \
112({ \
113 if((family) != AF_INET && (family) != AF_INET6) \
114 fatal$("GET_LEVELS: family not specified!"); \
115 (family) == AF_INET ? IPV4_LEVELS : IPV6_LEVELS; \
116 })
117#else
118#define GET_LEVELS(family) ({ (family)==AF_INET ? IPV4_LEVELS : IPV6_LEVELS; })
119#endif
