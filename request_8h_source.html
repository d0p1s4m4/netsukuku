127/*
128 * Request_table: It prevents requests flood and it is used in each connection.
129 * Each element of the "rq" array corresponds to a request; it (the element)
130 * keeps the number of requests served. If this number is equal
131 * to [REQUEST]_MAXRQ, the maximum of simultaneous requests is reached.
132 *
133 * Each element in rq_wait corresponds to a single request so it is formed by:
134 * { [REQUEST 0]_MAXRQ elements | [REQUEST 1]_MAXRQ elements | ... };
135 * rq_wait_idx keeps track of this but it must be initialized once with
136 * rq_wait_idx_init().
137 * Each element of rq_wait keeps the time when that request arrived.
138 * When the current time is >= [REQUEST]_WAIT+rq_wait, a new request is
139 * available and the corresponding request counter in "rq" is decremented.
140 */
141
142#define TOTAL_MAXRQ 31
143struct request_tbl {
144 u_char rq[TOTAL_REQUESTS];
145 time_t rq_wait[TOTAL_MAXRQ];
146};

153/*
154 * Each bit of this array corresponds to a request or a reply. If the bit is
155 * set, the request or reply will be dropped, otherwise it will be executed by
156 * pkts.c/pkt_exec().
157 */
158char filtered_op[TOTAL_OPS >> 3];
159#define OP_FILTER_DROP 1
160#define OP_FILTER_ALLOW 0
161#define op_filter_set(op) SET_BIT(filtered_op, (op))
162#define op_filter_clr(op) CLR_BIT(filtered_op, (op))
163#define op_filter_test(op) TEST_BIT(filtered_op, (op))
164#define op_filter_reset(bit) memset(filtered_op, (bit), sizeof(filtered_op))
